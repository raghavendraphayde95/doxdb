5   import javax.json.Json;
6   import javax.json.JsonArray;
7   import javax.json.JsonObject;
8   import javax.json.JsonValue;
9   import javax.ws.rs.client.ClientBuilder;
10  import javax.ws.rs.client.Entity;
11  import javax.ws.rs.client.Invocation.Builder;
12  import javax.ws.rs.core.MediaType;
13  
14  import net.trajano.commons.testing.ResourceUtil;
15  
16  /**
17   * This will call the REST API and load MOCK_DATA.json.
18   *
19   * @author Archimedes Trajano
20   */
21  public class DataLoader {
22  
23      public static void main(final String[] args) {
24  
25          final InputStream resourceAsStream = ResourceUtil.getResourceAsStream("MOCK_DATA.json");
26          final JsonArray collection = Json.createReader(resourceAsStream).readArray();
27          final Builder request = ClientBuilder.newClient().target("http://localhost:8080/doxdb-sample/V1/venue").request(MediaType.APPLICATION_JSON_TYPE);
28  
29          for (final JsonValue item : collection) {
30              final JsonObject record = (JsonObject) item;
31              final JsonObject jsonObject = Json.createObjectBuilder().add("name", record.get("last_name")).add("language", record.get("language")).build();
32  
33              request.post(Entity.entity(jsonObject, MediaType.APPLICATION_JSON_TYPE));
34          }
35  
36      }
37  }
