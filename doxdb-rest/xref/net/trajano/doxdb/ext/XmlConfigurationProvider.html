33  /**
34   * <p>
35   * This loads up the configuration from an XML file. The file is called
36   * <code>META-INF/dox.xml</code>. This class does not get loaded automatically
37   * as an EJB, instead it is expected that an EJB JAR file will declare this in
38   * the <code>META-INF/ejb-jar.xml</code> file to allow custom providers specific
39   * to the application if required. It is configured as follows:
40   * </p>
41   *
42   * <pre>
43   * &lt;enterprise-beans&gt;
44   *   &lt;session&gt;
45   *     &lt;ejb-name&gt;XmlConfigurationProvider&lt;/ejb-name&gt;
46   *     &lt;business-local&gt;net.trajano.doxdb.spi.ConfigurationProvider&lt;/business-local&gt;
47   *     &lt;ejb-class&gt;net.trajano.doxdb.spi.XmlConfigurationProvider&lt;/ejb-class&gt;
48   *     &lt;session-type&gt;Stateless&lt;/session-type&gt;
49   *   &lt;/session&gt;
50   * &lt;/enterprise-beans&gt;
51   * </pre>
52   *
53   * @author Archimedes Trajano
54   */
55  public class XmlConfigurationProvider implements
56      ConfigurationProvider {
57  
58      private final Map<String, SchemaType> currentSchemaMap = new HashMap<>();
59  
60      private final Map<String, DoxType> doxen = new HashMap<>();
61  
62      private final Map<String, String> indexMap;
63  
64      private final ConcurrentMap<String, JsonSchema> jsonSchemaMap = new ConcurrentHashMap<>();
65  
66      private final DoxPersistence persistenceConfig;
67  
68      public XmlConfigurationProvider() {
69  
70          try (final InputStream is = Thread.currentThread().getContextClassLoader().getResourceAsStream("META-INF/dox.xml")) {
71              final JAXBContext jaxb = JAXBContext.newInstance(DoxPersistence.class);
72              final Unmarshaller unmarshaller = jaxb.createUnmarshaller();
73              unmarshaller.setSchema(SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI).newSchema(getClass().getResource("/META-INF/xsd/dox.xsd")));
74              persistenceConfig = (DoxPersistence) unmarshaller.unmarshal(is);
75              indexMap = new ConcurrentHashMap<>(persistenceConfig.getIndex().size());
76              for (final IndexType indexType : persistenceConfig.getIndex()) {
77                  indexMap.put(indexType.getName(), indexType.getMappedName() == null ? indexType.getName() : indexType.getMappedName());
78              }
79  
80              for (final DoxType doxConfig : persistenceConfig.getDox()) {
81                  doxen.put(doxConfig.getName(), doxConfig);
82                  final SchemaType schema = doxConfig.getSchema().get(doxConfig.getSchema().size() - 1);
83                  currentSchemaMap.put(doxConfig.getName(), schema);
84                  //= jsonSchemaMap.get(schema.getLocation());
85                  final ValidationConfiguration cfg = ValidationConfiguration.newBuilder()
86                      .setDefaultVersion(SchemaVersion.DRAFTV4)
87                      .freeze();
88  
89                  final JsonSchema jsonSchema = JsonSchemaFactory.newBuilder()
90                      .setValidationConfiguration(cfg)
91                      .freeze()
92                      .getJsonSchema(JsonLoader.fromResource("/META-INF/schema/" + schema.getLocation()));
93                  jsonSchemaMap.putIfAbsent(schema.getLocation(), jsonSchema);
94  
95              }
96  
97          } catch (final IOException
98              | SAXException
99              | JAXBException
100             | ProcessingException e) {
101             throw new ExceptionInInitializerError(e);
102         }
103     }
104 
105     /**
106      * {@inheritDoc}
107      */
108     @Override
109     @Lock(LockType.READ)
110     public SchemaType getCollectionSchema(final String schemaName) {
111 
112         return currentSchemaMap.get(schemaName);
113     }
114 
115     /**
116      * {@inheritDoc}
117      */
118     @Override
119     @Lock(LockType.READ)
120     public JsonSchema getContentSchema(final String location) {
121 
122         return jsonSchemaMap.get(location);
123     }
124 
125     /**
126      * {@inheritDoc}
127      */
128     @Override
129     @Lock(LockType.READ)
130     public DoxType getDox(final String schemaName) {
131 
132         return doxen.get(schemaName);
133     }
134 
135     /**
136      * {@inheritDoc}
137      */
138     @Override
139     @Lock(LockType.READ)
140     public String getMappedIndex(final String name) {
141 
142         final String mappedName = indexMap.get(name);
143         if (mappedName == null) {
144             throw new PersistenceException("No index defined for " + name);
145         }
146         return mappedName;
147     }
148 
149     @Override
150     @Lock(LockType.READ)
151     public DoxPersistence getPersistenceConfig() {
152 
153         return persistenceConfig;
154     }
