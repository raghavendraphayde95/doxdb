13  @WebServlet("/import")
14  @MultipartConfig()
15  public class ImportServlet extends HttpServlet {
16  
17      /**
18       * serialVersionUID.
19       */
20      private static final long serialVersionUID = 7055873387528138394L;
21  
22      @Override
23      protected void doPost(final HttpServletRequest req,
24          final HttpServletResponse resp) throws ServletException,
25              IOException {
26  
27          long c = 0;
28          final InputStream cis = req.getPart("file").getInputStream();
29          while (cis.read() != -1) {
30              ++c;
31              if (c % 1024024 == 0) {
32                  resp.getWriter().print(". ");
33                  resp.getWriter().flush();
34              }
35          }
36          cis.close();
37          resp.getWriter().print(c);
38      }
39  }
