15      @Test
16      public void testJson() throws Exception {
17  
18          final ValidationConfiguration cfg = ValidationConfiguration.newBuilder()
19              .setDefaultVersion(SchemaVersion.DRAFTV4)
20              .freeze();
21  
22          final JsonSchema schema = JsonSchemaFactory.newBuilder()
23              .setValidationConfiguration(cfg)
24              .freeze()
25              .getJsonSchema(JsonLoader.fromResource("/schema/horse.json"));
26  
27          final ProcessingReport validate = schema.validate(JsonLoader.fromString("{\"name\":\"archie\"}"));
28          Assert.assertTrue(validate.toString(), validate.isSuccess());
29  
30      }
