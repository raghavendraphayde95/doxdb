78              @Override

81                  if ("myindex".equals(name)) {

83                  }
84                  throw new PersistenceException();

88              public DoxPersistence getPersistenceConfig() {
89  
90                  final DoxPersistence doxPersistence = new DoxPersistence();

92                  doxType.setName("horse");
93                  final SchemaType schema = new SchemaType();
94                  schema.setLocation("horse.json");
95                  schema.setVersion(1);
96                  doxType.getSchema().add(schema);
97                  doxPersistence.getDox().add(doxType);
98                  {
99                      final IndexType indexType = new IndexType();
100                     indexType.setName("myindex");
101                     indexType.setMappedName("test_myindex");
102                     doxPersistence.getIndex().add(indexType);
103                 }
104                 return doxPersistence;
105             }
