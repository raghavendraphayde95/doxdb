39      /**
40       * Create a unique ID for the search index record.
41       *
42       * @param view
43       * @return
44       */
45      public static String uid(final IndexView view) {
46  
47          return view.getIndex() + "\t" + view.getCollection() + "\t" + view.getDoxID();
48      }

54      /**
55       * {@inheritDoc}
56       */
57      @Override
58      @Asynchronous
59      public void addToIndex(final IndexView... indexViews) {
60  
61          final Bulk.Builder bulkBuilder = new Bulk.Builder();
62          for (final IndexView indexView : indexViews) {
63  
64              final Map<String, Object> map = new HashMap<>();
65              for (final Entry<String, Double> d : indexView.getDoubles()) {
66                  map.put(d.getKey(), d.getValue());
67              }
68              for (final Entry<String, Long> d : indexView.getLongs()) {
69                  map.put(d.getKey(), d.getValue());
70              }
71              for (final Entry<String, String> d : indexView.getStrings()) {
72                  map.put(d.getKey(), d.getValue());
73              }
74              final Map<String, Object> metaMap = new HashMap<>();
75              for (final Entry<String, String> d : indexView.getTexts()) {
76                  metaMap.put(d.getKey(), d.getValue());
77              }
78              metaMap.put("_text", indexView.getText());
79  
80              final Index index = new Index.Builder(map).index(configurationProvider.getMappedIndex(indexView.getIndex())).type(indexView.getCollection()).id(indexView.getDoxID().toString()).build();
81  
82              bulkBuilder.addAction(index);
83          }
84          final JestResult result = jestProvider.execute(bulkBuilder.build());
85          if (!result.isSucceeded()) {
86              throw new PersistenceException(result.getJsonString());
87          }
88  
89      }
90  
91      /**
92       * {@inheritDoc}
93       */
94      @Override
95      public void removeFromIndex(final String schemaName,
96          final DoxID doxID) {
97  
98          final Bulk.Builder bulkBuilder = new Bulk.Builder();
99          for (final IndexType indexType : configurationProvider.getPersistenceConfig().getIndex()) {
100 
101             final Delete del = new Delete.Builder(doxID.toString()).type(schemaName).index(configurationProvider.getMappedIndex(indexType.getName())).build();
102 
103             bulkBuilder.addAction(del);
104         }
105         jestProvider.execute(bulkBuilder.build());
106     }

118     /**
119      * {@inheritDoc}
120      */
121     @Override
122     @SuppressWarnings({
123         "rawtypes",
124         "deprecation"
125     })
126     public SearchResult search(final String index,
127         final String queryString,
128         final int limit,
129         final Integer fromDoc) {
130 
131         int from = 0;
132         if (fromDoc != null) {
133             from = fromDoc;
134         }
135 
136         final SearchResult result = new SearchResult();
137 
138         final JsonObjectBuilder sqsBuilder = Json.createObjectBuilder().add("query", queryString).add("default_operator", "and");
139         final JsonObjectBuilder qBuilder = Json.createObjectBuilder().add("simple_query_string", sqsBuilder);
140         final JsonObjectBuilder queryBuilder = Json.createObjectBuilder().add("size", limit).add("query", qBuilder).add("from", from);
141         final String query = queryBuilder.build().toString();
142 
143         final SearchResult2 esResults = new SearchResult2(jestProvider.execute(new Search.Builder(query).addIndex(index).build()));
144         result.setTotalHits(esResults.getTotal());
145 
146         final List<Map> hits = esResults.getSourceAsObjectList(Map.class);
147         result.setBottomDoc(Math.min(from + hits.size(), from + limit));
148         for (final Map hit : hits) {
149             final IndexView iv = new IndexView();
150             for (final Object key : hit.keySet()) {
151                 final Object value = hit.get(key);
152                 if (JestResult.ES_METADATA_ID.equals(key)) {
153                     iv.setDoxID(new DoxID((String) value));
154                 } else if (SearchResult2.ES_METADATA_TYPE.equals(key)) {
155                     iv.setCollection((String) value);
156                 } else if (value instanceof Double) {
157                     iv.setDouble(key.toString(), (double) value);
158                 } else if (value instanceof Long) {
159                     iv.setLong(key.toString(), (long) value);
160                 } else if (value instanceof String) {
161                     iv.setString((String) key, (String) value);
162                 }
163             }
164             result.addHit(iv);
165         }
166 
167         return result;
168 
169     }
