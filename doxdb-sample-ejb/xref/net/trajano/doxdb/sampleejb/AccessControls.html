23      Migrator.class

25  public class AccessControls implements
26      CollectionAccessControl,
27      Migrator,
28      Indexer {
29  
30      @Override
31      public byte[] buildAccessKey(final String collection,
32          final String json,
33          final String principal) {
34  
35          return "HELLO".getBytes();
36      }
37  
38      @Override
39      public IndexView[] buildIndexViews(final String collection,
40          final String json) {
41  
42          try {
43              if ("horse".equals(collection)) {
44                  final Horse horse = new ObjectMapper().readValue(json, Horse.class);
45  
46                  final IndexView indexView = new IndexView();
47                  indexView.setText("json", json);
48                  indexView.appendText(json);
49                  indexView.setString("collection", collection);
50                  indexView.setString("name", horse.getName());
51                  indexView.setIndex("myindex");
52  
53                  final IndexView indexView2 = new IndexView();
54                  indexView2.appendText(json);
55                  indexView2.setString("collection", collection);
56                  if (horse.getGender() == Gender.GELDING) {
57                      indexView2.setString("name", "*********************");
58                      indexView2.setText("name", horse.getName());
59                  } else {
60                      indexView2.setString("name", horse.getName());
61                  }
62                  indexView2.setIndex("maskedindex");
63  
64                  return new IndexView[] {
65                      indexView,
66                      indexView2
67                  };
68              } else if ("venue".equals(collection)) {
69                  final Venue venue = new ObjectMapper().readValue(json, Venue.class);
70  
71                  final IndexView indexView = new IndexView();
72                  indexView.setText("json", json);
73                  indexView.appendText(json);
74                  indexView.setString("collection", collection);
75                  indexView.setString("name", venue.getName());
76                  indexView.setLong("time", System.currentTimeMillis());
77                  indexView.setDouble("float", 156.6 / System.currentTimeMillis());
78                  indexView.setIndex("myindex");
79  
80                  final IndexView indexView2 = new IndexView();
81                  indexView2.appendText(json);
82                  indexView2.setString("collection", collection);
83                  if ("en".equals(venue.getLanguage())) {
84                      indexView2.setString("name", "*********************");
85                      indexView2.setText("name", venue.getName());
86                  } else {
87                      indexView2.setString("name", venue.getName());
88                  }
89                  indexView2.setIndex("maskedindex");
90  
91                  return new IndexView[] {
92                      indexView,
93                      indexView2
94                  };
95              } else {
96                  return new IndexView[0];
97              }
98  
99          } catch (final IOException e) {
100             throw new PersistenceException(e);
101         }
102     }
103
