139     private String capitalize(final String s) {
140 
141         if (s == null || s.isEmpty()) {
142             return s;
143         }
144         final char[] a = s.toCharArray();
145         a[0] = Character.toUpperCase(a[0]);
146         return new String(a);
147     }

170     @DELETE
171     @Path("{collection}/{id}")
172     public Response delete(@PathParam("collection") final String collection,
173         @PathParam("id") final DoxID doxid,
174         @QueryParam("v") final int version,
175         @QueryParam("extra") final String extraJson) {
176 
177         JsonObject extra;
178         if (extraJson != null) {
179             extra = Json.createReader(new StringReader(extraJson)).readObject();
180         } else {
181             extra = Json.createObjectBuilder().build();
182         }
183         dox.delete(collection, doxid, version, extra);
184         sessionManager.sendMessage("DELETE", doxid, collection, new Date());
185         return Response.noContent().build();
186     }

231      * In addition to the default operations, the lookup operations will also be
232      * provided as the string "getBy{lookupName}" for unique lookups. Note that
233      * lookupName will have it's first character capitalized.
234      *
235      * @param uriInfo

238     @GET
239     @Path("module.js")
240     @Produces("application/javascript")
241     public Response getAngularModule(final @Context UriInfo uriInfo) {
242 
243         final StreamingOutput out = new StreamingOutput() {
244 
245             @Override
246             public void write(final OutputStream os) throws IOException,
247                 WebApplicationException {
248 
249                 try (final PrintStream w = new PrintStream(os)) {
250                     w.println("\"use strict\";");
251                     w.print("(function(){");
252                     //                    w.print("function s(d){");
253                     //                    w.print("delete d._id;");
254                     //                    w.print("delete d._v;");
255                     //                    w.print("}");
256                     w.print("angular.module('doxdb',['ngResource'])");
257                     for (final CollectionType collectionType : dox.getConfiguration().getCollection()) {
258                         final String name = collectionType.getName();
259                         w.print(".factory('DoxDB" + name + "', ['$resource',function(r){");
260                         final String uri = uriInfo.getBaseUriBuilder().path(name).build() + "/:id?v=:version";
261                         w.print("return r('" + uri + "',{'id':'@_id','version':'@_version'},{");
262 
263                         final SchemaType currentSchemaType = collectionType.getSchema().get(collectionType.getSchema().size() - 1);
264 
265                         for (final LookupType LookupType : currentSchemaType.getUnique()) {
266                             final String lookupUri = uriInfo.getBaseUriBuilder().path(name).build() + "/" + LookupType.getName() + "/:lookupKey";
267                             w.print("'getBy" + capitalize(LookupType.getName()) + "':{method:'GET',url:'" + lookupUri + "'},");
268                         }
269 
270                         w.print("});");
271                         w.print("}])");
272                     }
273                     w.print(";})();");
274                 }
275             }
276         };
277 
278         return Response.ok(out).cacheControl(OK_CACHE).tag(dox.toString()).build();
279     }

281     /**
282      * Returns the schema document. It does a check to make sure each path
283      * segment contains a restricted set of characters.
284      *
285      * @param segments
286      *            path segments after the URL
287      * @return the schema document.
288      */
289     @GET
290     @Path("schema/{segments: .*}")
291     @Produces(RESPONSE_TYPE)
292     public Response getSchema(@PathParam("segments") final List<PathSegment> segments) {
293 
294         final UriBuilder b = UriBuilder.fromUri("schema");
295         for (final PathSegment segment : segments) {
296             final String pathSegment = segment.getPath();
297             if (".".equals(pathSegment) || "..".equals(pathSegment) || pathSegment.endsWith(".") || !pathSegment.matches("^[-A-Za-z0-9_\\.]+$")) {
298                 throw new WebApplicationException("invalid request");
299             }
300             b.path(pathSegment);
301         }
302         final URI relativize = UriBuilder.fromUri("schema").build().relativize(b.build());
303         if (relativize.isAbsolute()) {
304             throw new WebApplicationException("invalid request");
305         }
306 
307         final StreamingOutput out = new StreamingOutput() {
308 
309             @Override
310             public void write(final OutputStream os) throws IOException,
311                 WebApplicationException {
312 
313                 try (InputStream fis = dox.getSchema(relativize.toASCIIString())) {
314                     if (fis == null) {
315                         throw new NotFoundException();
316                     }
317                     int c = fis.read();
318                     while (c != -1) {
319                         os.write(c);
320                         c = fis.read();
321                     }
322                 }
323 
324             }
325         };
326         return Response.ok(out).encoding("UTF-8").build();
327     }

341     /**
342      * This will return either an array or a single JsobObject depending on
343      * whether the lookup is for unique or not.
344      *
345      * @param collectionName
346      *            collection name
347      * @param lookupName
348      *            lookup name (can be either unique or not)
349      * @param lookupKey
350      *            lookup key
351      * @return array or a single JsobObject depending on whether the lookup is
352      *         for unique or not.
353      */
354     @GET
355     @Path("{collectionName}/{lookupName}/{lookupKey}")
356     @Produces(RESPONSE_TYPE)
357     public Response lookup(@PathParam("collectionName") final String collectionName,
358         @PathParam("lookupName") final String lookupName,
359         @PathParam("lookupKey") final String lookupKey,
360         @Context final UriInfo uriInfo) {
361 
362         SchemaType schema = null;
363         for (final CollectionType collection : dox.getConfiguration().getCollection()) {
364             if (collection.getName().equals(collectionName)) {
365                 schema = collection.getSchema().get(collection.getSchema().size() - 1);
366                 break;
367             }
368         }
369         if (schema == null) {
370             return Response.status(Status.NOT_FOUND).build();
371         }
372 
373         for (final LookupType lookup : schema.getUnique()) {
374             if (lookup.getName().equals(lookupName)) {
375                 final DoxMeta meta = dox.readByUniqueLookup(collectionName, lookupName, lookupKey);
376                 if (meta == null) {
377                     return Response.status(Status.NOT_FOUND).type(MediaType.TEXT_PLAIN).entity("Dox not found").build();
378                 }
379                 final EntityTag entityTag = new EntityTag(String.valueOf(meta.getVersion()));
380                 final URI location = uriInfo.getBaseUriBuilder().path(collectionName).path(meta.getDoxId().toString()).build();
381                 return Response.seeOther(location).cacheControl(OK_CACHE).tag(entityTag).lastModified(meta.getLastUpdatedOn()).build();
382             }
383         }
384         for (final LookupType lookup : schema.getLookup()) {
385             if (lookup.getName().equals(lookupName)) {
386                 return Response.ok(dox.readByLookup(collectionName, lookupName, lookupKey)).build();
387             }
388         }
389         return Response.status(Status.NOT_FOUND).build();
390 
391     }

393     @GET
394     @Path("{collection}")
395     @Produces(RESPONSE_TYPE)
396     public Response readAll(@PathParam("collection") final String collection) {
397 
398         final String readAll = dox.readAll(collection);
399         if (readAll.charAt(0) == '[') {
400             return Response.ok(readAll).build();
401         }
402 
403         // The file is suffixed with the uncompressed size
404 
405         final File file = new File(readAll);
406         final long uncompressedSize = file.length();
407         final StreamingOutput out = new StreamingOutput() {
408 
409             @Override
410             public void write(final OutputStream os) throws IOException,
411                 WebApplicationException {
412 
413                 try (final InputStream fis = new BufferedInputStream(new FileInputStream(file))) {
414                     int c = fis.read();
415                     while (c != -1) {
416                         os.write(c);
417                         c = fis.read();
418                     }
419                 } finally {
420                     file.delete();
421                 }
422             }
423 
424         };
425 
426         return Response.ok(out).cacheControl(NO_CACHE).header("Content-Encoding", "gzip").header("Content-Length", uncompressedSize).build();
427 
428     }

463     /**
464      * This builds the search result object except for the link to the next
465      * page.
466      *
467      * @param uriInfo
468      *            context to build the URI
469      * @param results
470      *            the search results from the EJB
471      * @return JSON Object builder
472      */
473     private JsonObjectBuilder searchResultBuilder(final UriInfo uriInfo,
474         final SearchResult results) {
475 
476         final JsonObjectBuilder jsonBuilder;
477         final JsonArrayBuilder hitsBuilder = Json.createArrayBuilder();
478         for (final IndexView hit : results.getHits()) {
479             final JsonObjectBuilder hitBuilder = Json.createObjectBuilder();
480             final String id = hit.getDoxID().toString();
481             for (final Entry<String, BigDecimal> entry : hit.getNumbers()) {
482                 hitBuilder.add(entry.getKey(), entry.getValue());
483             }
484             for (final Entry<String, String> entry : hit.getStrings()) {
485                 hitBuilder.add(entry.getKey(), entry.getValue());
486             }
487             hitBuilder.add("_collection", hit.getCollection());
488             if (!hit.isMasked()) {
489                 hitBuilder.add("_id", id);
490                 hitBuilder.add("_url", uriInfo.getBaseUriBuilder().path(hit.getCollection()).path(id).build().toString());
491             }
492             hitsBuilder.add(hitBuilder);
493         }
494         jsonBuilder = Json.createObjectBuilder().add("totalHits", results.getTotalHits()).add("hits", hitsBuilder);
495         return jsonBuilder;
496     }

498     @GET
499     @Path("search/{index}")
500     @Produces(RESPONSE_TYPE)
501     public Response simpleSearch(@PathParam("index") final String index,
502         @QueryParam("q") final String queryString,
503         @QueryParam("f") final Integer from,
504         @Context final UriInfo uriInfo) {
505 
506         final SearchResult results = dox.search(index, queryString, 50, from);
507         final JsonObjectBuilder resultBuilder = searchResultBuilder(uriInfo, results);
508         if (results.getBottomDoc() != null) {
509             final String nextPage = uriInfo.getBaseUriBuilder().path("search").path(index).queryParam("q", queryString).queryParam("f", results.getBottomDoc()).build().toASCIIString();
510             resultBuilder.add("bottomDoc", results.getBottomDoc()).add("next", nextPage);
511         }
512         final JsonObject resultJson = resultBuilder.build();
513         return Response.ok(resultJson).cacheControl(NO_CACHE).build();
514     }
