85      @Path("{collection}")
86      @POST
87      @Consumes(MediaType.APPLICATION_JSON)
88      @Produces(RESPONSE_TYPE)
89      public Response create(@PathParam("collection") final String collection,
90          final String json) {
91  
92          final BsonDocument bson = BsonDocument.parse(json);
93          for (final String key : bson.keySet()) {
94              if (key.startsWith("_")) {
95                  bson.remove(key);
96              }
97          }
98  
99          final DoxMeta meta = dox.create(collection, bson);
100         sessionManager.sendMessage("CREATE", meta.getDoxId(), collection, meta.getLastUpdatedOn());
101         return Response.ok(meta.getContentJson()).lastModified(meta.getLastUpdatedOn()).build();
102     }

115     @GET
116     @Path("{collection}/{id}")
117     @Produces(RESPONSE_TYPE)
118     public Response get(@PathParam("collection") final String collection,
119         @PathParam("id") final DoxID doxid) {
120 
121         final DoxMeta meta = dox.read(collection, doxid);
122         if (meta == null) {
123             return Response.status(Status.NOT_FOUND).type(MediaType.TEXT_PLAIN).entity("Dox not found").build();
124         }
125         final EntityTag entityTag = new EntityTag(String.valueOf(meta.getVersion()));
126         return Response.ok(meta.getContentJson()).tag(entityTag).lastModified(meta.getLastUpdatedOn()).build();
127     }

129     /**
130      * Returns the schema document. It does a check to make sure each path
131      * segment contains a restricted set of characters.
132      *
133      * @param segments
134      *            path segments after the URL
135      * @return the schema document.
136      */
137     @GET
138     @Path("schema/{segments: .*}")
139     @Produces(RESPONSE_TYPE)
140     public Response getSchema(@PathParam("segments") final List<PathSegment> segments) {
141 
142         final UriBuilder b = UriBuilder.fromUri("schema");
143         for (final PathSegment segment : segments) {
144             final String pathSegment = segment.getPath();
145             if (".".equals(pathSegment) || "..".equals(pathSegment) || pathSegment.endsWith(".") || !pathSegment.matches("^[-A-Za-z0-9_\\.]+$")) {
146                 throw new WebApplicationException("invalid request");
147             }
148             b.path(pathSegment);
149         }
150         final URI relativize = UriBuilder.fromUri("schema").build().relativize(b.build());
151         if (relativize.isAbsolute()) {
152             throw new WebApplicationException("invalid request");
153         }
154 
155         final StreamingOutput out = new StreamingOutput() {
156 
157             @Override
158             public void write(final OutputStream os) throws IOException,
159                 WebApplicationException {
160 
161                 try (InputStream fis = dox.getSchema(relativize.toASCIIString())) {
162                     if (fis == null) {
163                         throw new NotFoundException();
164                     }
165                     int c = fis.read();
166                     while (c != -1) {
167                         os.write(c);
168                         c = fis.read();
169                     }
170                 }
171 
172             }
173         };
174         return Response.ok(out).encoding("UTF-8").build();
175     }

184     @GET
185     @Path("{collection}")
186     @Produces(RESPONSE_TYPE)
187     public Response readAll(@PathParam("collection") final String collection) {
188 
189         final String readAll = dox.readAll(collection);
190         if (readAll.charAt(0) == '[') {
191             return Response.ok(readAll).build();
192         }
193 
194         // The file is suffixed with the uncompressed size
195 
196         final File file = new File(readAll);
197         final long uncompressedSize = file.length();
198         final StreamingOutput out = new StreamingOutput() {
199 
200             @Override
201             public void write(final OutputStream os) throws IOException,
202                 WebApplicationException {
203 
204                 try (final InputStream fis = new BufferedInputStream(new FileInputStream(file))) {
205                     int c = fis.read();
206                     while (c != -1) {
207                         os.write(c);
208                         c = fis.read();
209                     }
210                 } finally {
211                     file.delete();
212                 }
213             }
214 
215         };
216 
217         return Response.ok(out).cacheControl(NO_CACHE).header("Content-Encoding", "gzip").header("Content-Length", uncompressedSize).build();
218 
219     }

230     private Response save(final String collection,
231         final String id,
232         final String json,
233         final int version) {
234 
235         final BsonDocument bson = BsonDocument.parse(json);
236 
237         final Iterator<Entry<String, BsonValue>> iterator = bson.entrySet().iterator();
238         while (iterator.hasNext()) {
239             final Entry<String, BsonValue> entry = iterator.next();
240             if (entry.getKey().startsWith("_")) {
241                 iterator.remove();
242             }
243         }
244 
245         final DoxMeta meta = dox.update(collection, new DoxID(id), bson, version);
246         sessionManager.sendMessage("UPDATE", meta.getDoxId(), collection, meta.getLastUpdatedOn());
247         return Response.ok(meta.getContentJson()).lastModified(meta.getLastUpdatedOn()).build();
248     }

265         @PathParam("idOrOp") final String idOrOp,
266         @QueryParam("v") final int version,
267         final String content) {
268 
269         if (idOrOp.startsWith("_")) {
270             return op(collection, idOrOp.substring(1), content);
271         } else {
272             return save(collection, idOrOp, content, version);
273         }

276     @GET
277     @Path("search/{index}")
278     @Produces(RESPONSE_TYPE)
279     public Response simpleSearch(@PathParam("index") final String index,
280         @QueryParam("q") final String queryString,
281         @QueryParam("f") final Integer from,
282         @Context final UriInfo uriInfo) {
283 
284         final SearchResult results = dox.search(index, queryString, 50, from);
285         //results.get
286         final JsonArrayBuilder hitsBuilder = Json.createArrayBuilder();
287         for (final IndexView hit : results.getHits()) {
288             final JsonObjectBuilder hitBuilder = Json.createObjectBuilder();
289             final String id = hit.getDoxID().toString();
290             for (final Entry<String, BigDecimal> entry : hit.getNumbers()) {
291                 hitBuilder.add(entry.getKey(), entry.getValue());
292             }
293             for (final Entry<String, String> entry : hit.getStrings()) {
294                 hitBuilder.add(entry.getKey(), entry.getValue());
295             }
296             hitBuilder.add("_collection", hit.getCollection());
297             if (!hit.isMasked()) {
298                 hitBuilder.add("_id", id);
299                 hitBuilder.add("_url", uriInfo.getBaseUriBuilder().path(hit.getCollection()).path(id).build().toString());
300             }
301             hitsBuilder.add(hitBuilder);
302         }
303         final JsonObjectBuilder jsonBuilder = Json.createObjectBuilder().add("totalHits", results.getTotalHits()).add("hits", hitsBuilder);
304         if (results.getBottomDoc() != null) {
305             final String nextPage = uriInfo.getBaseUriBuilder().path("search").path(index).queryParam("q", queryString).queryParam("f", results.getBottomDoc()).build().toASCIIString();
306             jsonBuilder.add("bottomDoc", results.getBottomDoc()).add("next", nextPage);
307         }
308         final JsonObject resultJson = jsonBuilder.build();
309         return Response.ok(resultJson).cacheControl(NO_CACHE).build();
310     }
