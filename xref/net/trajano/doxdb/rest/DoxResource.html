87      @Path("{collection}")
88      @POST
89      @Consumes(MediaType.APPLICATION_JSON)
90      @Produces(RESPONSE_TYPE)
91      public Response create(@PathParam("collection") final String collection,
92          final String json) {
93  
94          final BsonDocument bson = BsonDocument.parse(json);
95          for (final String key : bson.keySet()) {
96              if (key.startsWith("_")) {
97                  bson.remove(key);
98              }
99          }
100 
101         final DoxMeta meta = dox.create(collection, bson);
102         sessionManager.sendMessage("CREATE", meta.getDoxId(), collection, meta.getLastUpdatedOn());
103         return Response.ok(meta.getContentJson()).lastModified(meta.getLastUpdatedOn()).build();
104     }

117     @GET
118     @Path("{collection}/{id}")
119     @Produces(RESPONSE_TYPE)
120     public Response get(@PathParam("collection") final String collection,
121         @PathParam("id") final DoxID doxid) {
122 
123         final DoxMeta meta = dox.read(collection, doxid);
124         if (meta == null) {
125             return Response.status(Status.NOT_FOUND).type(MediaType.TEXT_PLAIN).entity("Dox not found").build();
126         }
127         final EntityTag entityTag = new EntityTag(String.valueOf(meta.getVersion()));
128         return Response.ok(meta.getContentJson()).tag(entityTag).lastModified(meta.getLastUpdatedOn()).build();
129     }

131     @GET
132     @Path("module.js")
133     @Produces("application/javascript")
134     public Response getAngularModule(final @Context UriInfo uriInfo) {
135 
136         final StreamingOutput out = new StreamingOutput() {
137 
138             @Override
139             public void write(final OutputStream os) throws IOException,
140                 WebApplicationException {
141 
142                 try (final PrintStream w = new PrintStream(os)) {
143                     w.println("\"use strict\";");
144                     w.print("(function() {");
145                     w.print("angular.module('doxdb',['ngResource'])");
146                     for (final DoxType doxType : dox.getConfiguration().getDox()) {
147                         final String name = doxType.getName();
148                         w.print(".factory('DoxDB" + name + "', ['$resource',function(r) {");
149                         final String uri = uriInfo.getBaseUriBuilder().path(name).build() + "/:id?v=:version";
150                         w.print("return r('" + uri + "',{ 'id':'@_id','version':'@_version' });");
151                         w.print("}])");
152                     }
153                     w.print(";})();");
154                 }
155             }
156         };
157 
158         return Response.ok(out).tag(dox.toString()).build();
159     }

161     /**
162      * Returns the schema document. It does a check to make sure each path
163      * segment contains a restricted set of characters.
164      *
165      * @param segments
166      *            path segments after the URL
167      * @return the schema document.
168      */
169     @GET
170     @Path("schema/{segments: .*}")
171     @Produces(RESPONSE_TYPE)
172     public Response getSchema(@PathParam("segments") final List<PathSegment> segments) {
173 
174         final UriBuilder b = UriBuilder.fromUri("schema");
175         for (final PathSegment segment : segments) {
176             final String pathSegment = segment.getPath();
177             if (".".equals(pathSegment) || "..".equals(pathSegment) || pathSegment.endsWith(".") || !pathSegment.matches("^[-A-Za-z0-9_\\.]+$")) {
178                 throw new WebApplicationException("invalid request");
179             }
180             b.path(pathSegment);
181         }
182         final URI relativize = UriBuilder.fromUri("schema").build().relativize(b.build());
183         if (relativize.isAbsolute()) {
184             throw new WebApplicationException("invalid request");
185         }
186 
187         final StreamingOutput out = new StreamingOutput() {
188 
189             @Override
190             public void write(final OutputStream os) throws IOException,
191                 WebApplicationException {
192 
193                 try (InputStream fis = dox.getSchema(relativize.toASCIIString())) {
194                     if (fis == null) {
195                         throw new NotFoundException();
196                     }
197                     int c = fis.read();
198                     while (c != -1) {
199                         os.write(c);
200                         c = fis.read();
201                     }
202                 }
203 
204             }
205         };
206         return Response.ok(out).encoding("UTF-8").build();
207     }

216     @GET
217     @Path("{collection}")
218     @Produces(RESPONSE_TYPE)
219     public Response readAll(@PathParam("collection") final String collection) {
220 
221         final String readAll = dox.readAll(collection);
222         if (readAll.charAt(0) == '[') {
223             return Response.ok(readAll).build();
224         }
225 
226         // The file is suffixed with the uncompressed size
227 
228         final File file = new File(readAll);
229         final long uncompressedSize = file.length();
230         final StreamingOutput out = new StreamingOutput() {
231 
232             @Override
233             public void write(final OutputStream os) throws IOException,
234                 WebApplicationException {
235 
236                 try (final InputStream fis = new BufferedInputStream(new FileInputStream(file))) {
237                     int c = fis.read();
238                     while (c != -1) {
239                         os.write(c);
240                         c = fis.read();
241                     }
242                 } finally {
243                     file.delete();
244                 }
245             }
246 
247         };
248 
249         return Response.ok(out).cacheControl(NO_CACHE).header("Content-Encoding", "gzip").header("Content-Length", uncompressedSize).build();
250 
251     }

262     private Response save(final String collection,
263         final String id,
264         final String json,
265         final int version) {
266 
267         final BsonDocument bson = BsonDocument.parse(json);
268 
269         final Iterator<Entry<String, BsonValue>> iterator = bson.entrySet().iterator();
270         while (iterator.hasNext()) {
271             final Entry<String, BsonValue> entry = iterator.next();
272             if (entry.getKey().startsWith("_")) {
273                 iterator.remove();
274             }
275         }
276 
277         final DoxMeta meta = dox.update(collection, new DoxID(id), bson, version);
278         sessionManager.sendMessage("UPDATE", meta.getDoxId(), collection, meta.getLastUpdatedOn());
279         return Response.ok(meta.getContentJson()).lastModified(meta.getLastUpdatedOn()).build();
280     }

298         @QueryParam("v") final int version,
299         final String content) {
300 
301         if (idOrOp.startsWith("_")) {

303         } else {
304             return save(collection, idOrOp, content, version);
305         }

308     @GET
309     @Path("search/{index}")
310     @Produces(RESPONSE_TYPE)
311     public Response simpleSearch(@PathParam("index") final String index,
312         @QueryParam("q") final String queryString,
313         @QueryParam("f") final Integer from,
314         @Context final UriInfo uriInfo) {
315 
316         final SearchResult results = dox.search(index, queryString, 50, from);
317         //results.get
318         final JsonArrayBuilder hitsBuilder = Json.createArrayBuilder();
319         for (final IndexView hit : results.getHits()) {
320             final JsonObjectBuilder hitBuilder = Json.createObjectBuilder();
321             final String id = hit.getDoxID().toString();
322             for (final Entry<String, BigDecimal> entry : hit.getNumbers()) {
323                 hitBuilder.add(entry.getKey(), entry.getValue());
324             }
325             for (final Entry<String, String> entry : hit.getStrings()) {
326                 hitBuilder.add(entry.getKey(), entry.getValue());
327             }
328             hitBuilder.add("_collection", hit.getCollection());
329             if (!hit.isMasked()) {
330                 hitBuilder.add("_id", id);
331                 hitBuilder.add("_url", uriInfo.getBaseUriBuilder().path(hit.getCollection()).path(id).build().toString());
332             }
333             hitsBuilder.add(hitBuilder);
334         }
335         final JsonObjectBuilder jsonBuilder = Json.createObjectBuilder().add("totalHits", results.getTotalHits()).add("hits", hitsBuilder);
336         if (results.getBottomDoc() != null) {
337             final String nextPage = uriInfo.getBaseUriBuilder().path("search").path(index).queryParam("q", queryString).queryParam("f", results.getBottomDoc()).build().toASCIIString();
338             jsonBuilder.add("bottomDoc", results.getBottomDoc()).add("next", nextPage);
339         }
340         final JsonObject resultJson = jsonBuilder.build();
341         return Response.ok(resultJson).cacheControl(NO_CACHE).build();
342     }
