25  /**
26   * Provides a secondary lookup table for the Dox data. It is used for simple
27   * lookups without resorting to using an elasticsearch index.
28   *
29   * @author Archimedes Trajano
30   */
31  @Entity
32  @Table(
33      indexes = {
34          @Index(columnList = "doxId"),
35          @Index(columnList = "collectionName,lookupName,lookupKey")
36  })
37  @NamedQueries({
38      @NamedQuery(name = DoxLookup.LOOKUP,
39          query = "select u.dox from DoxLookup u where u.collectionName = :collectionName and u.lookupName = :lookupName and u.lookupKey = :lookupKey",
40          lockMode = LockModeType.NONE),
41      @NamedQuery(name = DoxLookup.REMOVE_LOOKUP_FOR_DOX,
42          query = "delete from DoxLookup u where u.dox = :dox"),
43      @NamedQuery(name = DoxLookup.REMOVE_ALL,
44          query = "delete from DoxLookup "),
45      @NamedQuery(name = DoxLookup.UPDATE_LOOKUP_FOR_DOX,
46          query = "update DoxLookup u set u.lookupKey = :lookupKey where u.dox = :dox")
47  })
48  public class DoxLookup {
49  
50      public static final String COLLECTION_NAME = "collectionName";
51  
52      /**
53       * Named query {@value #LOOKUP}.
54       */
55      public static final String LOOKUP = "lookup";
56  
57      public static final String LOOKUP_KEY = "lookupKey";
58  
59      public static final String LOOKUP_NAME = "lookupName";
60  
61      public static final String REMOVE_ALL = "removeAllLookup";
62  
63      /**
64       * Named query {@value #REMOVE_LOOKUPFOR_DOX}.
65       */
66      public static final String REMOVE_LOOKUP_FOR_DOX = "removeLookupForDox";
67  
68      /**
69       * Named query {@value #UPDATE_LOOKUP_FOR_DOX}.
70       */
71      public static final String UPDATE_LOOKUP_FOR_DOX = "updateLookupForDox";
72  
73      /**
74       * Creates an list of {@link DoxUnique} based on the data from a {@link Dox}
75       * and the {@link SchemaType}.
76       *
77       * @param dox
78       *            dox
79       * @param schemaType
80       *            schema type
81       * @return list of {@link DoxUnique}.
82       */
83      public static List<DoxLookup> fromDox(final Dox dox,
84          final SchemaType schemaType) {
85  
86          final List<DoxLookup> a = new ArrayList<>(schemaType.getUnique().size());
87          for (final LookupType lookup : schemaType.getLookup()) {
88              final DoxLookup r = new DoxLookup();
89              r.collectionName = dox.getCollectionName();
90              r.dox = dox;
91              r.lookupName = lookup.getName();
92              r.lookupKey = JsonPath.compile(lookup.getPath()).read(dox.getJsonObject().toString());
93              a.add(r);
94          }
95  
96          return a;
97  
98      }
99
