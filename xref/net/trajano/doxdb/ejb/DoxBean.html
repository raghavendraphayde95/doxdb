77      /**
78       * This will create a new JsonObject with the _id and _version fields set.
79       * Also any top level values whose key starts with "_" is removed.
80       *
81       * @param jsonObject
82       *            object to decorate
83       * @param doxId
84       *            Dox ID
85       * @param version
86       *            optimistic locking version
87       * @return decorated JSOB object.
88       */
89      public static JsonObject decorateWithIdVersion(final JsonObject jsonObject,
90          final DoxID doxId,
91          final int version) {
92  
93          final JsonObjectBuilder b = Json.createObjectBuilder();
94          b.add("_id", doxId.toString());
95          b.add("_version", version);
96          for (final String key : jsonObject.keySet()) {
97              if (!key.startsWith("_")) {
98                  b.add(key, jsonObject.get(key));
99              }
100         }
101         return b.build();
102     }
103 
104     /**
105      * Extracts the extra info. Extra info is any property that starts with
106      * {@code _} except for {@code _id} and {@code _version}.
107      *
108      * @param jsonObject
109      * @return {@link JsonObject} containing only extra data.
110      */
111     private static JsonObject getExtra(final JsonObject jsonObject) {
112 
113         final JsonObjectBuilder b = Json.createObjectBuilder();
114         for (final String key : jsonObject.keySet()) {
115             if (key.startsWith("_") && !"_id".equals(key) && !"_version".equals(key) && jsonObject.getValueType() == JsonValue.ValueType.STRING) {
116                 b.add(key, jsonObject.get(key));
117             }
118         }
119         return b.build();
120     }

123      * This will create a new JsonObject with reserved properties removed.
124      * Reserved properties start with "_" including "_id" and "_version".
125      *
126      * @param jsonObject
127      *            object to decorate
128      * @param doxId
129      *            Dox ID
130      * @param version
131      *            optimistic locking version
132      * @return decorated JSOB object.
133      */
134     public static JsonObject sanitize(final JsonObject jsonObject) {
135 
136         final JsonObjectBuilder b = Json.createObjectBuilder();
137         for (final String key : jsonObject.keySet()) {
138             if (!key.startsWith("_")) {
139                 b.add(key, jsonObject.get(key));
140             }
141         }
142         return b.build();
143     }

183     @Override
184     public DoxMeta create(final String collectionName,
185         final JsonObject unsanitizedContent) {
186 
187         final Date ts = new Date();
188         final CollectionType config = configurationProvider.getCollection(collectionName);
189         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
190 
191         final JsonObject extra = getExtra(unsanitizedContent);
192         final JsonObject content = sanitize(unsanitizedContent);
193         validate(schema, content);
194 
195         final DoxID doxId = DoxID.generate();
196 
197         final String inputJson = content.toString();
198         final byte[] accessKey = collectionAccessControl.buildAccessKey(config.getName(), inputJson, ctx.getCallerPrincipal().getName());
199 
200         final Dox entity = new Dox();
201         entity.setDoxId(doxId);
202         entity.setContent(inputJson);
203         entity.setCreatedBy(ctx.getCallerPrincipal());
204         entity.setCreatedOn(ts);
205         entity.setLastUpdatedBy(ctx.getCallerPrincipal());
206         entity.setLastUpdatedOn(ts);
207         entity.setCollectionName(config.getName());
208         entity.setCollectionSchemaVersion(schema.getVersion());
209         entity.setAccessKey(accessKey);
210         entity.setVersion(1);
211 
212         em.persist(entity);
213 
214         for (final LookupType unique : schema.getUnique()) {
215             final String lookupKey = JsonPath.compile(unique.getPath()).read(inputJson);
216             final DoxUnique doxUnique = new DoxUnique();
217             doxUnique.setCollectionName(config.getName());
218             doxUnique.setDox(entity);
219             doxUnique.setLookupName(unique.getName());
220             doxUnique.setLookupKey(lookupKey);
221             em.persist(doxUnique);
222         }
223         for (final LookupType unique : schema.getLookup()) {
224             final String lookupKey = JsonPath.compile(unique.getPath()).read(inputJson);
225             final DoxLookup doxLookup = new DoxLookup();
226             doxLookup.setCollectionName(config.getName());
227             doxLookup.setDox(entity);
228             doxLookup.setLookupName(unique.getName());
229             doxLookup.setLookupKey(lookupKey);
230             em.persist(doxLookup);
231         }
232 
233         final IndexView[] indexViews = indexer.buildIndexViews(config.getName(), inputJson);
234         for (final IndexView indexView : indexViews) {
235             indexView.setCollection(config.getName());
236             indexView.setDoxID(doxId);
237         }
238         if (indexViews.length > 0) {
239             doxSearchBean.addToIndex(indexViews);
240         }
241         final DoxMeta meta = new DoxMeta();
242         meta.setCollectionName(collectionName);
243         meta.setAccessKey(accessKey);
244         meta.setLastUpdatedBy(ctx.getCallerPrincipal());
245         meta.setLastUpdatedOn(ts);
246         meta.setVersion(1);
247         meta.setDoxId(doxId);
248 
249         meta.setContentJson(content, doxId, 1);
250 
251         eventHandler.onRecordCreate(meta, content.toString(), extra);
252         return meta;
253     }

255     @Override
256     public boolean delete(final String collectionName,
257         final DoxID doxid,
258         final int version,
259         final JsonObject extraJson) {
260 
261         final Date ts = new Date();
262         final CollectionType config = configurationProvider.getCollection(collectionName);
263         final JsonObject extra = getExtra(extraJson);
264         final DoxMeta meta = readMetaAndLock(config.getName(), doxid, version);
265 
266         meta.getAccessKey();
267         // TODO check the security.
268 
269         final Dox toBeDeleted = em.find(Dox.class, meta.getId());
270         if (toBeDeleted == null) {
271             return false;
272         }
273         em.createNamedQuery(DoxUnique.REMOVE_UNIQUE_FOR_DOX).setParameter("dox", toBeDeleted).executeUpdate();
274         em.createNamedQuery(DoxLookup.REMOVE_LOOKUP_FOR_DOX).setParameter("dox", toBeDeleted).executeUpdate();
275 
276         final BsonDocument contentBson = toBeDeleted.getContent();
277         final DoxTombstone tombstone = toBeDeleted.buildTombstone(ctx.getCallerPrincipal(), ts);
278         em.persist(tombstone);
279         em.remove(toBeDeleted);
280 
281         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
282 
283         String contentJson = contentBson.toJson();
284         if (meta.getCollectionSchemaVersion() != schema.getVersion()) {
285             contentJson = migrator.migrate(collectionName, meta.getCollectionSchemaVersion(), schema.getVersion(), contentJson);
286         }
287 
288         doxSearchBean.removeFromIndex(config.getName(), doxid);
289         eventHandler.onRecordDelete(meta, contentJson, extra);
290         return true;
291 
292     }

294     /**
295      * Performs the update operation.
296      */
297     private DoxMeta doUpdate(final String collectionName,
298         final DoxID doxId,
299         final JsonObject unsanitizedContent,
300         final int version) {
301 
302         final Timestamp ts = new Timestamp(System.currentTimeMillis());
303         final CollectionType config = configurationProvider.getCollection(collectionName);
304         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
305 
306         final JsonObject extra = getExtra(unsanitizedContent);
307         final JsonObject content = sanitize(unsanitizedContent);
308         final String inputJson = content.toString();
309         validate(schema, inputJson);
310 
311         final DoxMeta meta = readMetaAndLock(config.getName(), doxId, version);
312         meta.incrementVersion();
313 
314         meta.getAccessKey();
315         // TODO check the security.
316 
317         final IndexView[] indexViews = indexer.buildIndexViews(config.getName(), inputJson);
318 
319         final byte[] accessKey = collectionAccessControl.buildAccessKey(config.getName(), inputJson, ctx.getCallerPrincipal().getName());
320 
321         final Dox e = em.find(Dox.class, meta.getId());
322         e.setLastUpdatedBy(ctx.getCallerPrincipal());
323         e.setLastUpdatedOn(ts);
324         e.setContent(content);
325         e.setAccessKey(accessKey);
326         em.persist(e);
327 
328         for (final LookupType unique : schema.getUnique()) {
329             final String lookupKey = JsonPath.compile(unique.getPath()).read(inputJson);
330             em.createNamedQuery(DoxUnique.UPDATE_UNIQUE_FOR_DOX).setParameter("dox", e).setParameter(DoxUnique.LOOKUP_KEY, lookupKey).executeUpdate();
331         }
332         for (final LookupType lookup : schema.getLookup()) {
333             final String lookupKey = JsonPath.compile(lookup.getPath()).read(inputJson);
334             em.createNamedQuery(DoxLookup.UPDATE_LOOKUP_FOR_DOX).setParameter("dox", e).setParameter(DoxUnique.LOOKUP_KEY, lookupKey).executeUpdate();
335         }
336 
337         for (final IndexView indexView : indexViews) {
338             indexView.setCollection(config.getName());
339             indexView.setDoxID(doxId);
340         }
341         doxSearchBean.addToIndex(indexViews);
342 
343         meta.setContentJson(content, doxId, e.getVersion());
344         eventHandler.onRecordUpdate(meta, e.getJsonContent(), extra);
345         return meta;
346 
347     }

364     @Override
365     public boolean isLocked(final String collectionName,
366         final DoxID doxId) {
367 
368         if (!configurationProvider.getCollection(collectionName).isLockable()) {
369             throw new PersistenceException(collectionName + " is not lockable");
370         }
371 
372         try {
373             em.createNamedQuery(DoxLock.READ_LOCK_BY_COLLECTION_NAME_DOX_ID)
374                 .setParameter(DoxLock.COLLECTION_NAME, collectionName)
375                 .setParameter(DoxLock.DOXID, doxId.toString())
376                 .getSingleResult();
377             return true;
378         } catch (final NoResultException e) {
379             return false;
380         }
381 
382     }

384     @Override
385     public int lock(final String collectionName,
386         final DoxID doxId) {
387 
388         if (!configurationProvider.getCollection(collectionName).isLockable()) {
389             throw new PersistenceException(collectionName + " is not lockable");
390         }
391 
392         final Date ts = new Date();
393         final DoxLock lock = new DoxLock();
394         lock.generateLockID();
395         lock.setLockedDox(em.createNamedQuery(Dox.READ_BY_COLLECTION_NAME_DOX_ID, Dox.class)
396             .setParameter(DoxLock.COLLECTION_NAME, collectionName)
397             .setParameter(DoxLock.DOXID, doxId.toString())
398             .getSingleResult());
399         lock.setLockedBy(ctx.getCallerPrincipal());
400         lock.setLockedOn(ts);
401 
402         em.persist(lock);
403         return lock.getLockId();
404     }

414     @Override
415     public DoxMeta read(final String collectionName,
416         final DoxID doxid) {
417 
418         final CollectionType config = configurationProvider.getCollection(collectionName);
419         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
420 
421         final DoxMeta meta;
422         try {
423             meta = em.createNamedQuery(Dox.READ_META_BY_COLLECTION_NAME_DOX_ID, DoxMeta.class).setParameter("doxId", doxid.toString()).setParameter("collectionName", config.getName()).getSingleResult();
424         } catch (final NoResultException e) {
425             return null;
426         }
427         meta.getAccessKey();
428         // TODO check the security.
429 
430         if (meta.getCollectionSchemaVersion() != schema.getVersion()) {
431             final Dox e = em.find(Dox.class, meta.getId(), LockModeType.OPTIMISTIC_FORCE_INCREMENT);
432             final String contentJson = migrator.migrate(collectionName, e.getCollectionSchemaVersion(), schema.getVersion(), e.getJsonContent());
433             meta.setCollectionName(collectionName);
434             meta.setCollectionSchemaVersion(schema.getVersion());
435             e.setCollectionSchemaVersion(schema.getVersion());
436             e.setContent(contentJson);
437             em.persist(e);
438             em.flush();
439             em.refresh(e);
440             final JsonObject content = e.getJsonObject();
441             meta.setContentJson(content, meta.getDoxId(), e.getVersion());
442         } else {
443             final Dox e = em.find(Dox.class, meta.getId(), LockModeType.OPTIMISTIC);
444             final JsonObject content = e.getJsonObject();
445             meta.setContentJson(content, meta.getDoxId(), meta.getVersion());
446         }
447         eventHandler.onRecordRead(ctx.getCallerPrincipal(), collectionName, doxid, meta.getContentJson());
448         return meta;
449 
450     }

462             } catch (final IOException e) {

468             throw new PersistenceException("Not supported");
469         }
470

473     /**
474      * Reads all records in a collection and writes it to a file.
475      *
476      * @param collectionName
477      * @return
478      * @throws IOException
479      */
480     private String readAllToFile(final String collectionName) throws IOException {
481 
482         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
483 
484         final File f = File.createTempFile("doxdb", collectionName);
485 
486         try (final Writer os = new OutputStreamWriter(new GZIPOutputStream(new FileOutputStream(f)), "UTF-8")) {
487             os.write('[');
488 
489             final List<Dox> results = em.createNamedQuery(Dox.READ_ALL_BY_COLLECTION_NAME, Dox.class).setParameter("collectionName", collectionName).getResultList();
490             final Iterator<Dox> i = results.iterator();
491             while (i.hasNext()) {
492 
493                 final Dox result = i.next();
494                 final boolean last = !i.hasNext();
495                 result.getAccessKey();
496                 // TODO check security
497                 if (result.getCollectionSchemaVersion() != schema.getVersion()) {
498                     migrator.migrate(collectionName, result.getCollectionSchemaVersion(), schema.getVersion(), result.getJsonContent());
499                     // queue migrate later?
500                 } else {
501                     os.write(decorateWithIdVersion(result.getJsonObject(), result.getDoxId(), result.getVersion()).toString());
502                     if (!last) {
503                         os.write(',');
504                     }
505                 }
506 
507             }
508             os.write(']');
509         }
510         return f.getCanonicalPath();
511 
512     }

514     private String readAllToString(final String collectionName) {
515 
516         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
517 
518         final StringBuilder b = new StringBuilder("[");
519 
520         final List<Dox> results = em.createNamedQuery(Dox.READ_ALL_BY_COLLECTION_NAME, Dox.class).setParameter(Dox.COLLECTION_NAME, collectionName).getResultList();
521         for (final Dox result : results) {
522 
523             result.getAccessKey();
524             // TODO check security
525             if (result.getCollectionSchemaVersion() != schema.getVersion()) {
526                 migrator.migrate(collectionName, result.getCollectionSchemaVersion(), schema.getVersion(), result.getJsonContent());
527                 // queue migrate later?
528             } else {
529                 b.append(decorateWithIdVersion(result.getJsonObject(), result.getDoxId(), result.getVersion()).toString());
530                 b.append(',');
531             }
532 
533         }
534         if (b.length() > 1) {
535             b.replace(b.length() - 1, b.length(), "]");
536         } else {
537             b.append(']');
538         }
539         return b.toString();

604     @Override
605     @Asynchronous
606     public void reindex() {
607 
608         doxSearchBean.reset();
609         // TODO this will do everything in one transaction which can kill the database.  What could be done is the
610         // reindexing can be done in chunks and let an MDB do the process
611         for (final CollectionType config : configurationProvider.getPersistenceConfig().getCollection()) {
612 
613             em.createNamedQuery(DoxUnique.REMOVE_ALL)
614                 .executeUpdate();
615 
616             em.createNamedQuery(DoxLookup.REMOVE_ALL)
617                 .executeUpdate();
618 
619             final SchemaType schemaType = config.getSchema().get(config.getSchema().size() - 1);
620 
621             final List<IndexView> indexViews = new LinkedList<>();
622             for (final Dox e : em.createNamedQuery(Dox.READ_ALL_BY_COLLECTION_NAME, Dox.class).setParameter(Dox.COLLECTION_NAME, config.getName()).getResultList()) {
623 
624                 for (final DoxUnique doxUnique : DoxUnique.fromDox(e, schemaType)) {
625                     em.persist(doxUnique);
626                 }
627                 for (final DoxLookup doxLookup : DoxLookup.fromDox(e, schemaType)) {
628                     em.persist(doxLookup);
629                 }
630                 // TODO later
631                 //                rs.updateBytes(3, collectionAccessControl.buildAccessKey(config.getName(), json, new DoxPrincipal(rs.getString(4))));
632                 final IndexView[] indexViewBuilt = indexer.buildIndexViews(config.getName(), e.getJsonContent());
633                 for (final IndexView indexView : indexViewBuilt) {
634                     indexView.setCollection(e.getCollectionName());
635                     indexView.setDoxID(e.getDoxId());
636                     indexViews.add(indexView);
637                 }
638 
639             }
640             doxSearchBean.addToIndex(indexViews.toArray(new IndexView[0]));
641 
642         }
643 
644     }

726     /**
727      * {@inheritDoc}
728      */
729     @Override
730     public void unlock(final String collectionName,
731         final DoxID doxId,
732         final int lockId) {
733 
734         if (!configurationProvider.getCollection(collectionName).isLockable()) {
735             throw new PersistenceException(collectionName + " is not lockable");
736         }
737 
738         em.createNamedQuery(DoxLock.REMOVE_LOCK_BY_COLLECTION_NAME_DOX_ID_LOCK_ID)
739             .setParameter(DoxLock.COLLECTION_NAME, collectionName)
740             .setParameter(DoxLock.DOXID, doxId.toString())
741             .setParameter(DoxLock.LOCKID, lockId).executeUpdate();
742 
743     }

760     /**
761      * {@inheritDoc}
762      */
763     @Override
764     public DoxMeta update(final String collectionName,
765         final DoxID doxId,
766         final JsonObject contents,
767         final int version,
768         final int lockId) {
769 
770         if (!configurationProvider.getCollection(collectionName).isLockable()) {
771             throw new PersistenceException(collectionName + " is not lockable");
772         }
773         verifyLockedBy(collectionName, doxId, lockId);
774         return doUpdate(collectionName, doxId, contents, version);
775     }

784     /**
785      * Performs JSON validation using a schema
786      *
787      * @param schema
788      *            schema
789      * @param json
790      *            json to validate.
791      */
792     private void validate(final SchemaType schema,
793         final String json) {
794 
795         try {
796 
797             final JsonSchema jsonSchema = configurationProvider.getContentSchema(schema.getLocation());
798 
799             final ProcessingReport validate = jsonSchema.validate(JsonLoader.fromString(json));
800             if (!validate.isSuccess()) {
801                 throw new ValidationException(validate.toString());
802             }
803         } catch (ProcessingException
804             | IOException e) {
805             throw new PersistenceException(e);
806         }
807     }

809     /**
810      * Check if a record is locked by the given lock ID.
811      *
812      * @param collectionName
813      *            collection name
814      * @param doxId
815      *            Dox ID
816      * @param lockId
817      *            lock ID
818      */
819     private void verifyLockedBy(final String collectionName,
820         final DoxID doxId,
821         final int lockId) {

824             .setParameter(DoxLock.COLLECTION_NAME, collectionName)
825             .setParameter(DoxLock.DOXID, doxId.toString())
826             .setParameter(DoxLock.LOCKID, lockId).getSingleResult();
827     }
