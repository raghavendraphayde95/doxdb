75      /**
76       * This will create a new JsonObject with the _id and _version fields set.
77       * Also any top level values whose key starts with "_" is removed.
78       *
79       * @param jsonObject
80       *            object to decorate
81       * @param doxId
82       *            Dox ID
83       * @param version
84       *            optimistic locking version
85       * @return decorated JSOB object.
86       */
87      public static JsonObject decorateWithIdVersion(final JsonObject jsonObject,
88          final DoxID doxId,
89          final int version) {
90  
91          final JsonObjectBuilder b = Json.createObjectBuilder();
92          b.add("_id", doxId.toString());
93          b.add("_version", version);
94          for (final String key : jsonObject.keySet()) {
95              if (!key.startsWith("_")) {
96                  b.add(key, jsonObject.get(key));
97              }
98          }
99          return b.build();
100     }
101 
102     /**
103      * This will create a new JsonObject with reserved properties removed.
104      * Reserved properties start with "_" including "_id" and "_version".
105      *
106      * @param jsonObject
107      *            object to decorate
108      * @param doxId
109      *            Dox ID
110      * @param version
111      *            optimistic locking version
112      * @return decorated JSOB object.
113      */
114     public static JsonObject sanitize(final JsonObject jsonObject) {
115 
116         final JsonObjectBuilder b = Json.createObjectBuilder();
117         for (final String key : jsonObject.keySet()) {
118             if (!key.startsWith("_")) {
119                 b.add(key, jsonObject.get(key));
120             }
121         }
122         return b.build();
123     }

156     @Override
157     public DoxMeta create(final String schemaName,
158         final JsonObject unsanitizedContent) {
159 
160         final Date ts = new Date();
161         final CollectionType config = configurationProvider.getDox(schemaName);
162         final SchemaType schema = configurationProvider.getCollectionSchema(schemaName);
163 
164         final JsonObject content = sanitize(unsanitizedContent);
165         validate(schema, content);
166 
167         final DoxID doxId = DoxID.generate();
168 
169         final String inputJson = content.toString();
170         final byte[] accessKey = collectionAccessControl.buildAccessKey(config.getName(), inputJson, ctx.getCallerPrincipal().getName());
171 
172         final Dox entity = new Dox();
173         entity.setDoxId(doxId);
174         entity.setContent(inputJson);
175         entity.setCreatedBy(ctx.getCallerPrincipal());
176         entity.setCreatedOn(ts);
177         entity.setLastUpdatedBy(ctx.getCallerPrincipal());
178         entity.setLastUpdatedOn(ts);
179         entity.setCollectionName(config.getName());
180         entity.setCollectionSchemaVersion(schema.getVersion());
181         entity.setAccessKey(accessKey);
182         entity.setVersion(1);
183 
184         em.persist(entity);
185 
186         for (final LookupType unique : schema.getUnique()) {
187             final String lookupKey = JsonPath.compile(unique.getPath()).read(inputJson);
188             final DoxUnique doxUnique = new DoxUnique();
189             doxUnique.setCollectionName(config.getName());
190             doxUnique.setDox(entity);
191             doxUnique.setLookupName(unique.getName());
192             doxUnique.setLookupKey(lookupKey);
193             em.persist(doxUnique);
194         }
195         for (final LookupType unique : schema.getLookup()) {
196             final String lookupKey = JsonPath.compile(unique.getPath()).read(inputJson);
197             final DoxLookup doxLookup = new DoxLookup();
198             doxLookup.setCollectionName(config.getName());
199             doxLookup.setDox(entity);
200             doxLookup.setLookupName(unique.getName());
201             doxLookup.setLookupKey(lookupKey);
202             em.persist(doxLookup);
203         }
204 
205         final IndexView[] indexViews = indexer.buildIndexViews(config.getName(), inputJson);
206         for (final IndexView indexView : indexViews) {
207             indexView.setCollection(config.getName());
208             indexView.setDoxID(doxId);
209         }
210         if (indexViews.length > 0) {
211             doxSearchBean.addToIndex(indexViews);
212         }
213         final DoxMeta meta = new DoxMeta();
214         meta.setAccessKey(accessKey);
215         meta.setLastUpdatedOn(ts);
216         meta.setVersion(1);
217         meta.setDoxId(doxId);
218 
219         meta.setContentJson(content, doxId, 1);
220 
221         eventHandler.onRecordCreate(config.getName(), doxId, inputJson);
222         return meta;
223     }

225     @Override
226     public boolean delete(final String collectionName,
227         final DoxID doxid,
228         final int version) {
229 
230         final Date ts = new Date();
231         final CollectionType config = configurationProvider.getDox(collectionName);
232         final DoxMeta meta = readMetaAndLock(config.getName(), doxid, version);
233 
234         meta.getAccessKey();
235         // TODO check the security.
236 
237         final Dox toBeDeleted = em.find(Dox.class, meta.getId());
238         if (toBeDeleted == null) {
239             return false;
240         }
241         em.createNamedQuery(DoxUnique.REMOVE_UNIQUE_FOR_DOX).setParameter("dox", toBeDeleted).executeUpdate();
242         em.createNamedQuery(DoxLookup.REMOVE_LOOKUP_FOR_DOX).setParameter("dox", toBeDeleted).executeUpdate();
243 
244         final BsonDocument contentBson = toBeDeleted.getContent();
245         final DoxTombstone tombstone = toBeDeleted.buildTombstone(ctx.getCallerPrincipal(), ts);
246         em.persist(tombstone);
247         em.remove(toBeDeleted);
248 
249         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
250 
251         String contentJson = contentBson.toJson();
252         if (meta.getCollectionSchemaVersion() != schema.getVersion()) {
253             contentJson = migrator.migrate(collectionName, meta.getCollectionSchemaVersion(), schema.getVersion(), contentJson);
254         }
255 
256         doxSearchBean.removeFromIndex(config.getName(), doxid);
257         eventHandler.onRecordDelete(config.getName(), doxid, contentJson);
258         return true;
259 
260     }

282     @Override
283     public DoxMeta read(final String collectionName,
284         final DoxID doxid) {
285 
286         final CollectionType config = configurationProvider.getDox(collectionName);
287         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
288 
289         final DoxMeta meta = em.createNamedQuery(Dox.READ_META_BY_SCHEMA_NAME_DOX_ID, DoxMeta.class).setParameter("doxId", doxid.toString()).setParameter("collectionName", config.getName()).getSingleResult();
290 
291         meta.getAccessKey();
292         // TODO check the security.
293 
294         if (meta.getCollectionSchemaVersion() != schema.getVersion()) {
295             final Dox e = em.find(Dox.class, meta.getId(), LockModeType.OPTIMISTIC_FORCE_INCREMENT);
296             final String contentJson = migrator.migrate(collectionName, e.getCollectionSchemaVersion(), schema.getVersion(), e.getJsonContent());
297             meta.setCollectionSchemaVersion(schema.getVersion());
298             e.setCollectionSchemaVersion(schema.getVersion());
299             e.setContent(contentJson);
300             em.persist(e);
301             em.flush();
302             em.refresh(e);
303             final JsonObject content = e.getJsonObject();
304             meta.setContentJson(content, meta.getDoxId(), e.getVersion());
305         } else {
306             final Dox e = em.find(Dox.class, meta.getId(), LockModeType.OPTIMISTIC);
307             final JsonObject content = e.getJsonObject();
308             meta.setContentJson(content, meta.getDoxId(), meta.getVersion());
309         }
310 
311         eventHandler.onRecordCreate(config.getName(), doxid, meta.getContentJson());
312         return meta;
313 
314     }

316     @Override
317     public String readAll(final String collectionName) {
318 
319         final CollectionType config = configurationProvider.getDox(collectionName);
320         if (config.getReadAll() == ReadAllType.FILE) {
321             try {
322                 return readAllToFile(config.getName());
323             } catch (final IOException e) {
324                 throw new PersistenceException(e);
325             }
326         } else if (config.getReadAll() == ReadAllType.MEMORY) {
327             return readAllToString(config.getName());
328         } else {
329             throw new PersistenceException("Not supported");
330         }
331 
332     }

334     private String readAllToFile(final String collectionName) throws IOException {
335 
336         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
337 
338         final File f = File.createTempFile("doxdb", collectionName);
339 
340         try (final Writer os = new OutputStreamWriter(new GZIPOutputStream(new FileOutputStream(f)), "UTF-8")) {
341             os.write('[');
342 
343             final List<Dox> results = em.createNamedQuery(Dox.READ_ALL_BY_SCHEMA_NAME, Dox.class).setParameter("collectionName", collectionName).getResultList();
344             final Iterator<Dox> i = results.iterator();
345             while (i.hasNext()) {
346 
347                 final Dox result = i.next();
348                 final boolean last = !i.hasNext();
349                 result.getAccessKey();
350                 // TODO check security
351                 if (result.getCollectionSchemaVersion() != schema.getVersion()) {
352                     migrator.migrate(collectionName, result.getCollectionSchemaVersion(), schema.getVersion(), result.getJsonContent());
353                     // queue migrate later?
354                 } else {
355                     os.write(decorateWithIdVersion(result.getJsonObject(), result.getDoxId(), result.getVersion()).toString());
356                     if (!last) {
357                         os.write(',');
358                     }
359                 }
360 
361             }
362             os.write(']');
363         }
364         return f.getCanonicalPath();
365 
366     }

368     private String readAllToString(final String collectionName) {
369 
370         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
371 
372         final StringBuilder b = new StringBuilder("[");
373 
374         final List<Dox> results = em.createNamedQuery(Dox.READ_ALL_BY_SCHEMA_NAME, Dox.class).setParameter(Dox.COLLECTION_NAME, collectionName).getResultList();
375         for (final Dox result : results) {
376 
377             result.getAccessKey();
378             // TODO check security
379             if (result.getCollectionSchemaVersion() != schema.getVersion()) {
380                 migrator.migrate(collectionName, result.getCollectionSchemaVersion(), schema.getVersion(), result.getJsonContent());
381                 // queue migrate later?
382             } else {
383                 b.append(decorateWithIdVersion(result.getJsonObject(), result.getDoxId(), result.getVersion()).toString());
384                 b.append(',');
385             }
386 
387         }
388         if (b.length() > 1) {
389             b.replace(b.length() - 1, b.length(), "]");
390         } else {
391             b.append(']');
392         }
393         return b.toString();
394 
395     }

401

408

410

416                 migrator.migrate(collectionName, result.getCollectionSchemaVersion(), schema.getVersion(), result.getJsonContent());
417                 // queue migrate later?
418             } else {
419                 b.add(decorateWithIdVersion(result.getJsonObject(), result.getDoxId(), result.getVersion()));

456     @Asynchronous
457     public void reindex() {
458 
459         doxSearchBean.reset();
460         // TODO this will do everything in one transaction which can kill the database.  What could be done is the
461         // reindexing can be done in chunks and let an MDB do the process
462         for (final CollectionType config : configurationProvider.getPersistenceConfig().getDox()) {
463 
464             final List<IndexView> indexViews = new LinkedList<>();
465             for (final Dox e : em.createNamedQuery(Dox.READ_ALL_BY_SCHEMA_NAME, Dox.class).setParameter("schemaName", config.getName()).getResultList()) {
466 
467                 // TODO later
468                 //                rs.updateBytes(3, collectionAccessControl.buildAccessKey(config.getName(), json, new DoxPrincipal(rs.getString(4))));
469                 final IndexView[] indexViewBuilt = indexer.buildIndexViews(config.getName(), e.getJsonContent());
470                 for (final IndexView indexView : indexViewBuilt) {
471                     indexView.setCollection(e.getCollectionName());
472                     indexView.setDoxID(e.getDoxId());
473                     indexViews.add(indexView);
474                 }
475 
476             }
477             doxSearchBean.addToIndex(indexViews.toArray(new IndexView[0]));
478 
479         }
480

566     @Override
567     public DoxMeta update(final String collectionName,
568         final DoxID doxId,
569         final JsonObject unsanitizedContent,
570         final int version) {
571 
572         final Timestamp ts = new Timestamp(System.currentTimeMillis());
573         final CollectionType config = configurationProvider.getDox(collectionName);
574         final SchemaType schema = configurationProvider.getCollectionSchema(collectionName);
575 
576         final JsonObject content = sanitize(unsanitizedContent);
577         final String inputJson = content.toString();
578         validate(schema, inputJson);
579 
580         final DoxMeta meta = readMetaAndLock(config.getName(), doxId, version);
581         meta.incrementVersion();
582 
583         meta.getAccessKey();
584         // TODO check the security.
585 
586         final IndexView[] indexViews = indexer.buildIndexViews(config.getName(), inputJson);
587 
588         final byte[] accessKey = collectionAccessControl.buildAccessKey(config.getName(), inputJson, ctx.getCallerPrincipal().getName());
589 
590         final Dox e = em.find(Dox.class, meta.getId());
591         e.setLastUpdatedBy(ctx.getCallerPrincipal());
592         e.setLastUpdatedOn(ts);
593         e.setContent(content);
594         e.setAccessKey(accessKey);
595         em.persist(e);
596 
597         for (final LookupType unique : schema.getUnique()) {
598             final String lookupKey = JsonPath.compile(unique.getPath()).read(inputJson);
599             em.createNamedQuery(DoxUnique.UPDATE_UNIQUE_FOR_DOX).setParameter("dox", e).setParameter(DoxUnique.LOOKUP_KEY, lookupKey).executeUpdate();
600         }
601         for (final LookupType lookup : schema.getLookup()) {
602             final String lookupKey = JsonPath.compile(lookup.getPath()).read(inputJson);
603             em.createNamedQuery(DoxLookup.UPDATE_LOOKUP_FOR_DOX).setParameter("dox", e).setParameter(DoxUnique.LOOKUP_KEY, lookupKey).executeUpdate();
604         }
605 
606         for (final IndexView indexView : indexViews) {
607             indexView.setCollection(config.getName());
608             indexView.setDoxID(doxId);
609         }
610         doxSearchBean.addToIndex(indexViews);
611 
612         eventHandler.onRecordUpdate(config.getName(), doxId, e.getJsonContent());
613         meta.setContentJson(content, doxId, e.getVersion());
614         return meta;
615 
616     }

625     /**
626      * Performs JSON validation using a schema
627      *
628      * @param schema
629      *            schema
630      * @param json
631      *            json to validate.
632      */
633     private void validate(final SchemaType schema,
634         final String json) {
635 
636         try {
637 
638             final JsonSchema jsonSchema = configurationProvider.getContentSchema(schema.getLocation());
639 
640             final ProcessingReport validate = jsonSchema.validate(JsonLoader.fromString(json));
641             if (!validate.isSuccess()) {
642                 throw new ValidationException(validate.toString());
643             }
644         } catch (ProcessingException
645             | IOException e) {
646             throw new PersistenceException(e);
647         }
648     }
